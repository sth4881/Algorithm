## 백준 18405 빙산 (골드 4)

### 문제링크
- https://www.acmicpc.net/problem/2573

### 알고리즘
- 72번부터 106번째 줄까지 코드 확인

### 소스코드
```java
import java.io.*;
import java.util.*;

public class BOJ2573 {
    static int N, M, answer;
    static int[][] grid;
    static boolean[][] visited;
    static int[] dx = { -1, 0, 1, 0 };
    static int[] dy = { 0, 1, 0, -1 };
    static List<Iceberg> list;
    static class Iceberg {
        int x;
        int y;
        int melt;
        Iceberg(int x, int y, int melt) {
            this.x = x;
            this.y = y;
            this.melt = melt;
        }
    }
    static boolean check(int nx, int ny) {
        return nx>=0 && nx<N && ny>=0 && ny<M;
    }
    static void dfs(int x, int y) {
        visited[x][y] = true;
        for(int i=0; i<4; i++) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if(check(nx, ny) && grid[nx][ny]>0 && !visited[nx][ny]) {
                dfs(nx, ny);
            }
        }
    }
    static void checkIceberg(int x, int y) {
        int cnt = 0;
        for(int i=0; i<4; i++) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if(check(nx, ny) && grid[nx][ny]==0) cnt++;
        }
        if(cnt>0) list.add(new Iceberg(x, y, cnt));
    }
    public static void main(String[] args) {
        try {
            BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
            BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));

            StringTokenizer token = new StringTokenizer(br.readLine());
            N = Integer.parseInt(token.nextToken());
            M = Integer.parseInt(token.nextToken());
            grid = new int[N][M];
            visited = new boolean[N][M];

            list = new ArrayList<>();
            for(int i=0; i<N; i++) {
                token = new StringTokenizer(br.readLine());
                for(int j=0; j<M; j++) {
                    grid[i][j] = Integer.parseInt(token.nextToken());
                }
            }

            int cnt = 0;
            while(true) {
                // 빙산이 분리됐는지 확인하기 위한 DFS
                for(int i=0; i<N; i++) {
                    for(int j=0; j<M; j++) {
                        if(grid[i][j]>0 && !visited[i][j]) {
                            dfs(i, j);
                            cnt++;
                        }
                    }
                }
                if(cnt!=1) break; // 빙산이 분리됐거나 분리되지 않은 상태로 녹았으면 break

                // 녹이려는 빙산 색출
                for(int i=0; i<N; i++) {
                    for(int j=0; j<M; j++) {
                        if(grid[i][j]>0) checkIceberg(i, j);
                    }
                }

                // 색출한 빙산 한번에 녹이기
                for(Iceberg ice : list) {
                    grid[ice.x][ice.y] -= ice.melt;
                    if(grid[ice.x][ice.y]<0) grid[ice.x][ice.y] = 0;
                }

                // 초기화
                for(boolean[] item : visited)
                    Arrays.fill(item, false);
                list.clear();
                answer++;
                cnt = 0;
            }
            if(cnt!=0) bw.write(answer+"\n");
            else bw.write(0+"\n");
            bw.close();
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
```
